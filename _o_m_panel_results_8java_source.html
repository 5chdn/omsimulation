00340       ObjectContainer db4o = Db4oEmbedded.openFile(
00341           Db4oEmbedded.newConfiguration(), getOmsFile());
00342       ObjectSet<OMSimulation> result = db4o.queryByExample(OMSimulation.class);
00343       OMSimulation found;
00344       tmpUpdate("Refreshing list... ", 2);
00345       tmpUpdate("Adding items... ", 3);
00346       for (int i = 0; i < result.size(); i++) {
00347         double perc = (double) i / (double) result.size() * 100.0 + 3.0;
00348         while (perc > 99) {
00349           perc--;
00350         }
00351         found = (OMSimulation) result.next();
00352         comboBoxSimulations.addItem(found);
00353         tmpUpdate("Added: " + found, (int) perc);
00354       }
00355       tmpUpdate("Finished. ", 100);
00356       db4o.close();
