00621     private String[] parseCSV(String fileName) throws IOException {
00622       String[] tmpArray = new String[65536];
00623       String[] csv = null;
00624       tmpUpdate("Trying to read CSV-File '" + fileName + "'.", 2);
00625       try {
00626         FileInputStream fileInput = new FileInputStream(fileName);
00627         DataInputStream dataInput = new DataInputStream(fileInput);
00628         InputStreamReader inputReader = new InputStreamReader(dataInput);
00629         BufferedReader buffReader = new BufferedReader(inputReader);
00630         tmpUpdate("Read CSV-File with success, trying to "
00631             + "parse line by line.", 2);
00632         int valueCount = 0;
00633         while ((tmpArray[valueCount] = buffReader.readLine()) != null) {
00634           valueCount++;
00635         }
00636         valueCount--;
00637         if (valueCount >= 168) {
00638           if (valueCount <= 1008) {
00639             csv = new String[valueCount + 1];
00640             for (int x = 0; x <= valueCount; x++) {
00641               csv[x] = tmpArray[x];
00642               logOnly(csv[x], 2);
00643             }
00644             setValueCount(valueCount);
00645             tmpUpdate("Parsed " + valueCount + " lines with success.", 2);
00646           } else {
00647             csv = new String[0];
00648             tmpUpdate("Error: " + valueCount
00649                 + " are too many data sets. Aborting.", 0);
00650             tmpUpdate(
00651                 "Error: Make sure you have at maximum six weeks of records (<= 1008).",
00652                 0);
00653           }
00654         } else {
00655           csv = new String[0];
00656           tmpUpdate("Error: " + valueCount
00657               + " are not enough data sets. Aborting.", 0);
00658           tmpUpdate(
00659               "Error: Make sure you have at least one week of records (>= 168).",
00660               0);
00661         }
00662         buffReader.close();
00663         inputReader.close();
00664         dataInput.close();
00665         fileInput.close();
00666       } catch (Exception e) {
00667         csv = new String[0];
00668         tmpUpdate("Error: " + e.getMessage(), 0);
00669         tmpUpdate("Error: Failed to read CSV-File" + fileName + ".", 0);
00670         e.printStackTrace();

00691     private OMRoom[] parseRooms(String[] csv, double detectionLimit)
00692         throws IOException {
00693       tmpUpdate("Trying to parse rooms.", 3);
00694       OMRoom[] rooms;
00695       try {
00696         if (csv.length > 1) {
00697           String header = csv[0];
00698           String[] roomId = header.split("\\;");
00699           int roomCount = roomId.length;
00700           for (int i = 1; i < roomCount; i++) {
00701             tmpUpdate("Found room: " + roomId[i], 3);
00702             if (roomId[i].isEmpty()) {
00703               tmpUpdate(
00704                   "Warning: malformed room identifier. Using \"m0\" for misc.",
00705                   3);
00706               roomId[i] = "m0";
00707             }
00708           }
00709           roomCount--;
00710           setRoomCount(roomCount);
00711           tmpUpdate("Parsed " + roomCount + " rooms with success.", 3);
00712           rooms = new OMRoom[roomCount];
00713           int valueCount = getValueCount();
00714           double[] values[] = new double[roomCount][valueCount];
00715           tmpUpdate("Trying to collect values for each room.", 3);
00716           String[] tmpValues = roomId;
00717           int tmpLength = tmpValues.length;
00718           boolean success = true;
00719           for (int i = 1; i < csv.length; i++) {
00720             tmpValues = csv[i].split("\\;");
00721             boolean isFirstDigit = Character.isDigit(csv[i].charAt(0));
00722             boolean isLastDigit = Character.isDigit(csv[i].charAt(csv[i]
00723                 .length() - 1));
00724             if (isFirstDigit && isLastDigit) {
00725               if (tmpValues.length == tmpLength) {
00726                 for (int j = 1; j < tmpValues.length; j++) {
00727                   int x = j - 1;
00728                   int y = i - 1;
00729                   if (tmpValues[j].isEmpty()) {
00730                     tmpUpdate("Warning: Empty string. Using value 0.", 4);
00731                     values[x][y] = 0.0;
00732                   } else {
00733                     values[x][y] = (double) Integer.parseInt(tmpValues[j]);
00734                   }
00735                   if (values[x][y] < detectionLimit) {
00736                     tmpUpdate("Warning: Value " + i + " for room " + roomId[j]
00737                         + ": '" + tmpValues[j]
00738                         + "' below detection limit. Using half of the limit ("
00739                         + (detectionLimit / 2.0) + ").", 4);
00740                     values[x][y] = detectionLimit / 2.0;
00741                   }
00742                   logOnly("Parsed value " + i + " for room " + roomId[j] + ": "
00743                       + values[x][y], 4);
00744                 }
00745                 tmpLength = tmpValues.length;
00746               } else {
00747                 tmpUpdate("Error: Malformed CSV-file. Aborting.", 0);
00748                 i = csv.length + 1;
00749                 success = false;
00750               }
00751             } else {
00752               tmpUpdate("Error: Malformed CSV-line: " + csv[i], 0);
00753               i = csv.length + 1;
00754               success = false;
00755             }
00756           }
00757           if (success) {
00758             for (int k = 1; k <= roomCount; k++) {
00759               int z = k - 1;
00760               rooms[z] = new OMRoom(roomId[k], values[z]);
00761             }
00762             tmpUpdate("Successfully collected values for each room.", 5);
00763           } else {
00764             rooms = new OMRoom[0];
00765             tmpUpdate("Error: Check your CSV-File.", 0);
00766           }
00767         } else {
00768           rooms = new OMRoom[0];
00769           tmpUpdate("Error: No data records found.", 0);
00770           tmpUpdate("Error: Check your CSV-File.", 0);
00771         }
00772       } catch (Exception e) {
00773         rooms = new OMRoom[0];
00774         tmpUpdate("Error: " + e.getMessage(), 0);
00775         tmpUpdate("Error: Failed to parse rooms.", 0);
00776         e.printStackTrace();
00777       }
00778       return rooms;
00779     }

00800     private OMBuilding separateRooms(String name, Date date, OMRoom[] rooms)
00801         throws IOException {
00802       tmpUpdate("Trying to separate rooms.", 6);
00803       int cellarCount = 0;
00804       int normalCount = 0;
00805       int miscCount = 0;
00806       int roomCount = getRoomCount();
00807       OMBuilding building = new OMBuilding();
00808       if (rooms.length > 0) {
00809         for (int i = 0; i < roomCount; i++) {
00810           if (rooms[i].getType() == OMRoomType.Cellar) {
00811             cellarCount++;
00812           } else {
00813             if (rooms[i].getType() == OMRoomType.Room) {
00814               normalCount++;
00815             } else {
00816               miscCount++;
00817             }
00818           }
00819         }
00820         OMRoom[] cellars = new OMRoom[cellarCount];
00821         OMRoom[] normals = new OMRoom[normalCount];
00822         OMRoom[] miscs = new OMRoom[miscCount];
00823         tmpUpdate("Found " + normalCount + " normal rooms, " + cellarCount
00824             + " cellars and " + miscCount + " miscellaneous rooms.", 6);
00825         if (normalCount >= 3 && cellarCount >= 1) {
00826           if (normalCount <= 8 && cellarCount <= 4) {
00827             if (roomCount == cellarCount + normalCount + miscCount) {
00828               tmpUpdate("Separation test succeeded, separating rooms.", 6);
00829               cellarCount = 0;
00830               normalCount = 0;
00831               miscCount = 0;
00832               for (int j = 0; j < roomCount; j++) {
00833                 if (rooms[j].getType() == OMRoomType.Cellar) {
00834                   cellars[cellarCount] = rooms[j];
00835                   cellarCount++;
00836                 } else {
00837                   if (rooms[j].getType() == OMRoomType.Room) {
00838                     normals[normalCount] = rooms[j];
00839                     normalCount++;
00840                   } else {
00841                     miscs[miscCount] = rooms[j];
00842                     miscCount++;
00843                   }
00844                 }
00845               }
00846               tmpUpdate(
00847                   "Setting up rooms by types. This may take a few minutes.", 7);
00848               progressBarImport.setIndeterminate(true);
00849               int valueCount = getValueCount();
00850               building = new OMBuilding(name, date, roomCount, valueCount,
00851                   normals, cellars, miscs);
00852               progressBarImport.setIndeterminate(false);
00853               tmpUpdate("Done. Finished setting up rooms.", 68);
00854             } else {
00855               tmpUpdate("Error: Separation test failed. Malformed input.", 0);
00856             }
00857           } else {
00858             tmpUpdate(
00859                 "Error: Separation test failed. Too many rooms or cellars.", 0);
00860             tmpUpdate(
00861                 "Error: Make sure you don't have more than 8 rooms and 4 cellars.",
00862                 0);
00863           }
00864         } else {
00865           tmpUpdate(
00866               "Error: Separation test failed. Not enough rooms or cellars.", 0);
00867           tmpUpdate("Error: Make sure you have at least 3 rooms and 1 cellar.",
00868               0);
00869         }
00870       } else {
00871         tmpUpdate(
00872             "Error: Separation test failed. Not enough rooms or cellars.", 0);
