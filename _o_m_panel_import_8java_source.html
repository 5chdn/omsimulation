00654     private OMRoom[] parseRooms(String[] csv, double detectionLimit)
00655         throws IOException {
00656       tmpUpdate("Trying to parse rooms.", 3);
00657       OMRoom[] rooms;
00658       try {
00659         if (csv.length > 1) {
00660           String header = csv[0];
00661           String[] roomId = header.split("\\;");
00662           int roomCount = roomId.length;
00663           for (int i = 1; i < roomCount; i++) {
00664             tmpUpdate("Found room: " + roomId[i], 3);
00665             if (roomId[i].isEmpty()) {
00666               tmpUpdate(
00667                   "Warning: malformed room identifier. Using \"m0\" for misc.",
00668                   3);
00669               roomId[i] = "m0";
00670             }
00671           }
00672           roomCount--;
00673           setRoomCount(roomCount);
00674           tmpUpdate("Parsed " + roomCount + " rooms with success.", 3);
00675           rooms = new OMRoom[roomCount];
00676           int valueCount = getValueCount();
00677           double[] values[] = new double[roomCount][valueCount];
00678           tmpUpdate("Trying to collect values for each room.", 3);
00679           String[] tmpValues = roomId;
00680           int tmpLength = tmpValues.length;
00681           boolean success = true;
00682           for (int i = 1; i < csv.length; i++) {
00683             tmpValues = csv[i].split("\\;");
00684             boolean isFirstDigit = Character.isDigit(csv[i].charAt(0));
00685             boolean isLastDigit = Character.isDigit(csv[i].charAt(csv[i]
00686                 .length() - 1));
00687             if (isFirstDigit && isLastDigit) {
00688               if (tmpValues.length == tmpLength) {
00689                 for (int j = 1; j < tmpValues.length; j++) {
00690                   int x = j - 1;
00691                   int y = i - 1;
00692                   if (tmpValues[j].isEmpty()) {
00693                     tmpUpdate("Warning: Empty string. Using value 0.", 4);
00694                     values[x][y] = 0.0;
00695                   } else {
00696                     values[x][y] = (double) Integer.parseInt(tmpValues[j]);
00697                   }
00698                   if (values[x][y] < detectionLimit) {
00699                     tmpUpdate("Warning: Value " + i + " for room " + roomId[j]
00700                         + ": '" + tmpValues[j]
00701                         + "' below detection limit. Using half of the limit ("
00702                         + (detectionLimit / 2.0) + ").", 4);
00703                     values[x][y] = detectionLimit / 2.0;
00704                   }
00705                   logOnly("Parsed value " + i + " for room " + roomId[j] + ": "
00706                       + values[x][y], 4);
00707                 }
00708                 tmpLength = tmpValues.length;
00709               } else {
00710                 tmpUpdate("Error: Malformed CSV-file. Aborting.", 0);
00711                 i = csv.length + 1;
00712                 success = false;
00713               }
00714             } else {
00715               tmpUpdate("Error: Malformed CSV-line: " + csv[i], 0);
00716               i = csv.length + 1;
00717               success = false;
00718             }
00719           }
00720           if (success) {
00721             for (int k = 1; k <= roomCount; k++) {
00722               int z = k - 1;
00723               rooms[z] = new OMRoom(roomId[k], values[z]);
00724             }
00725             tmpUpdate("Successfully collected values for each room.", 5);
00726           } else {
00727             rooms = new OMRoom[0];
00728             tmpUpdate("Error: Check your CSV-File.", 0);
00729           }
00730         } else {
00731           rooms = new OMRoom[0];
00732           tmpUpdate("Error: No data records found.", 0);
00733           tmpUpdate("Error: Check your CSV-File.", 0);
00734         }
00735       } catch (Exception e) {
00736         rooms = new OMRoom[0];
00737         tmpUpdate("Error: " + e.getMessage(), 0);
00738         tmpUpdate("Error: Failed to parse rooms.", 0);
00739       }
00740       return rooms;
00741     }

00762     private OMBuilding separateRooms(String name, Date date, OMRoom[] rooms)
00763         throws IOException {
00764       tmpUpdate("Trying to separate rooms.", 6);
00765       int cellarCount = 0;
00766       int normalCount = 0;
00767       int miscCount = 0;
00768       int roomCount = getRoomCount();
00769       OMBuilding building = new OMBuilding();
00770       if (rooms.length > 0) {
00771         for (int i = 0; i < roomCount; i++) {
00772           if (rooms[i].getType() == OMRoomType.Cellar) {
00773             cellarCount++;
00774           } else {
00775             if (rooms[i].getType() == OMRoomType.Room) {
00776               normalCount++;
00777             } else {
00778               miscCount++;
00779             }
00780           }
00781         }
00782         OMRoom[] cellars = new OMRoom[cellarCount];
00783         OMRoom[] normals = new OMRoom[normalCount];
00784         OMRoom[] miscs = new OMRoom[miscCount];
00785         tmpUpdate("Found " + normalCount + " normal rooms, " + cellarCount
00786             + " cellars and " + miscCount + " miscellaneous rooms.", 6);
00787         if (normalCount >= 3 && cellarCount >= 1) {
00788           if (normalCount <= 8 && cellarCount <= 4) {
00789             if (roomCount == cellarCount + normalCount + miscCount) {
00790               tmpUpdate("Separation test succeeded, separating rooms.", 6);
00791               cellarCount = 0;
00792               normalCount = 0;
00793               miscCount = 0;
00794               for (int j = 0; j < roomCount; j++) {
00795                 if (rooms[j].getType() == OMRoomType.Cellar) {
00796                   cellars[cellarCount] = rooms[j];
00797                   cellarCount++;
00798                 } else {
00799                   if (rooms[j].getType() == OMRoomType.Room) {
00800                     normals[normalCount] = rooms[j];
00801                     normalCount++;
00802                   } else {
00803                     miscs[miscCount] = rooms[j];
00804                     miscCount++;
00805                   }
00806                 }
00807               }
00808               tmpUpdate(
00809                   "Setting up rooms by types. This may take a few minutes.", 7);
00810               progressBarImport.setIndeterminate(true);
00811               int valueCount = getValueCount();
00812               building = new OMBuilding(name, date, roomCount, valueCount,
00813                   normals, cellars, miscs);
00814               progressBarImport.setIndeterminate(false);
00815               tmpUpdate("Done. Finished setting up rooms.", 68);
00816             } else {
00817               tmpUpdate("Error: Separation test failed. Malformed input.", 0);
00818             }
00819           } else {
00820             tmpUpdate(
00821                 "Error: Separation test failed. Too many rooms or cellars.", 0);
00822             tmpUpdate(
00823                 "Error: Make sure you don't have more than 8 rooms and 4 cellars.",
00824                 0);
00825           }
00826         } else {
00827           tmpUpdate(
00828               "Error: Separation test failed. Not enough rooms or cellars.", 0);
00829           tmpUpdate("Error: Make sure you have at least 3 rooms and 1 cellar.",
00830               0);
00831         }
00832       } else {
00833         tmpUpdate(
00834             "Error: Separation test failed. Not enough rooms or cellars.", 0);
