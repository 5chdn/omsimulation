02114     private String timeLeft(double perc) {
02115       String strFormat = "#";
02116       double timePassed = (System.currentTimeMillis() - getStart()) / 1000;
02117       double time = (timePassed / perc * 100.0) - timePassed;
02118       String unit = " seconds.";
02119       if (time > 100) {
02120         strFormat = "#";
02121         time = time / 60.0;
02122         unit = " minutes.";
02123       }
02124       if (time > 100) {
02125         strFormat = "#.#";
02126         time = time / 60.0;
02127         unit = " hours.";
02128       }
02129       final double days = time / 60.0 / 60.0 / 24.0;
02130       if (time > 48) {
02131         strFormat = "#.##";
02132         time = days;
02133         unit = " days.";
02134       }
02135       if (time > 50) {
02136         strFormat = "#.##";
02137         time = days / 28.0;
02138         unit = " months.";
02139       }
02140       if (time > 20) {
02141         strFormat = "#.##";
02142         time = days / 365.2424;
02143         unit = " years.";
02144       }
02145       DecimalFormat decFormat = new DecimalFormat(strFormat);
02146       return decFormat.format(time) + unit;
02147     }
